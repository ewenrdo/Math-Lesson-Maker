\subsubsection{Signature d'une forme quadratique}

\definition{
    Soit $q$ une forme quadratique sur $E$.\\
    On dit que $q$ est \textbf{positive} si $q(x) \geq 0$ pour tout $x \in E$, et que $q$ est \textbf{nÃĐgative} si $q(x) \leq 0$ pour tout $x \in E$.\\

\definition{

    On appelle \textbf{signature} de $q$ le couple d'entiers $(p,r)$ tel que $q = \ell_1^2 + \cdots + \ell_p^2 - \ell_{p+1}^2 - \cdots - \ell_{p+r}^2$ pour $\ell_1, \ldots, \ell_{p+r}$ des formes linÃĐaires indÃĐpendantes.\\
