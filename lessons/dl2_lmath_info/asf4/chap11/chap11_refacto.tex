\docTitle{Chapitre 11 : Fonctions de plusieurs variables}

\section{Continuité}

Soit $\mathcal{U}$ un ouvert de $\mathbb{R}^n$. Une fonction de plusieurs variables est définie par :
\[ f : \mathcal{U} \to \mathbb{R}, \quad x \mapsto f(x) = f(x_1, \dots, x_n) \]

\definition{
    Soit $x_0 \in \mathcal{U}$. On dit que $f$ est continue en $x_0$ si :
    \[ \forall \epsilon > 0, \exists \alpha > 0, \forall x \in \mathcal{U}, \quad \|x - x_0\| \le \alpha \Rightarrow |f(x) - f(x_0)| < \epsilon \]
}

\definition{
    $f$ est continue \textit{(séquentiellement)} en $x_0$ si et seulement si pour toute suite $(x_n)_{n \in \mathbb{N}}$ de $\mathcal{U}$ convergeant vers $x_0$, on a $f(x_n) \xrightarrow[n \to +\infty]{} f(x_0)$.
}

\remark{Par équivalence des normes en dimension finie, la notion de continuité ne dépend pas de la norme choisie dans $\mathbb{R}^n$.}

\attention{Il ne suffit pas que les applications partielles soient continues pour que $f$ soit continue.}

\cexample{
    Soit $f(x,y) = \begin{cases} \frac{xy}{x^2+y^2} & \text{si } (x,y) \neq (0,0) \\ 0 & \text{sinon} \end{cases}$.\\
    Les applications partielles $x \mapsto f(x,0)$ et $y \mapsto f(0,y)$ sont nulles donc continues. Cependant, en approchant par la droite $y=x$ :
    \[ f(x,x) = \frac{x^2}{x^2+x^2} = \frac{1}{2} \xrightarrow[x \to 0]{} \frac{1}{2} \neq f(0,0) \]
    $f$ n'est pas continue en $(0,0)$.
}

\section{Dérivabilité et Différentiabilité}

\subsection{Dérivées partielles et directionnelles}

\definition{
    Soit $v \in \mathbb{R}^n \setminus \{0\}$ et $x_0 \in \mathcal{U}$. On dit que $f$ admet une \textbf{dérivée en $x_0$ dans la direction $v$} si la limite suivante existe :
    \[ \lim_{t \to 0} \frac{f(x_0 + tv) - f(x_0)}{t} \]
}

\definition{
    On appelle \textbf{dérivées partielles} en $x_0$ les dérivées dans les directions de la base canonique $(e_1, \dots, e_n)$. On note :
    \[ \frac{\partial f}{\partial x_i}(x_0) = \lim_{t \to 0} \frac{f(x_0 + te_i) - f(x_0)}{t} \]
}

\cexample{
    La fonction $f(x,y) = \frac{xy}{x^2+y^2}$ (prolongée par $0$) admet des dérivées partielles nulles en $(0,0)$ car $f(x,0)=0$ et $f(0,y)=0$. On remarque ainsi que \textbf{l'existence de dérivées partielles n'implique pas la continuité}.
}

\subsection{La notion de différentiabilité}

La différentiabilité est la "vraie" notion de dérivabilité pour les fonctions de plusieurs variables, car elle assure une approximation linéaire globale.

\definition{
    On dit que $f$ est \textbf{différentiable} en $x_0$ s'il existe une forme linéaire $L : \mathbb{R}^n \to \mathbb{R}$ telle que :
    \[ f(x_0+h) = f(x_0) + L(h) + o(\|h\|) \]
    Cette forme linéaire est unique, on la note $Df(x_0)$ (la \textbf{différentielle}).
}

\theorem{Proposition}{Lien différentielle et dérivées partielles}{false}{
    Si $f$ est différentiable en $x_0$, alors elle admet des dérivées partielles dans toutes les directions et :
    \[ Df(x_0) \cdot h = \sum_{i=1}^n h_i \frac{\partial f}{\partial x_i}(x_0) \]
}

\definition{
    On appelle \textbf{gradient} de $f$ en $x_0$, noté $\nabla f(x_0)$, le vecteur représentant la différentielle dans la base canonique pour le produit scalaire usuel :
    \[ \nabla f(x_0) = \begin{pmatrix} \frac{\partial f}{\partial x_1}(x_0) \\ \vdots \\ \frac{\partial f}{\partial x_n}(x_0) \end{pmatrix} \]
}

\section{Fonctions de classe $\mathcal{C}^k$ et second ordre}

\definition{
    On dit que $f$ est de classe $\mathcal{C}^1$ sur $\mathcal{U}$ si elle est différentiable sur $\mathcal{U}$ et que l'application $x \mapsto Df(x)$ est continue.
}

\theorem{Proposition}{Caractérisation de la classe $\mathcal{C}^1$}{false}{
    $f$ est de classe $\mathcal{C}^1$ sur $\mathcal{U}$ si et seulement si ses dérivées partielles existent et sont \textbf{continues} sur $\mathcal{U}$.
}

\subsection{Différentielle seconde et Hessienne}

\definition{
    $f$ est $2$-fois différentiable en $x_0$ si $f$ est différentiable au voisinage de $x_0$ et $x \mapsto Df(x)$ est différentiable en $x_0$. On note $D^2f(x_0)$ cette différentielle seconde (forme bilinéaire).
}

\definition{
    On appelle \textbf{Matrice Hessienne} de $f$ en $x_0$ la matrice de $D^2f(x_0)$ dans la base canonique :
    \[
        Hess(f)(x_0) = \left( \frac{\partial^2 f}{\partial x_i \partial x_j}(x_0) \right)_{1 \le i,j \le n}
    \]
}

\theorem{Théorème de Schwarz}{}{false}{
    Si $f$ est deux fois différentiable en $x_0$ (ou de classe $\mathcal{C}^2$), alors la matrice Hessienne est \textbf{symétrique} :
    \[ \frac{\partial^2 f}{\partial x_i \partial x_j} = \frac{\partial^2 f}{\partial x_j \partial x_i} \]
}

\theorem{Théorème de Taylor-Young (ordre 2)}{}{false}{
    Si $f$ est deux fois différentiable en $x_0$ :
    \[ f(x_0 + h) = f(x_0) + Df(x_0) \cdot h + \frac{1}{2} D^2f(x_0)(h,h) + o(\|h\|^2) \]
}

\section{Recherche d'extrema}

\definition{
    On dit que $x_0$ est un \textbf{point critique} de $f$ si $Df(x_0) = 0$ (soit $\nabla f(x_0) = 0$).
}

\subsection{Conditions du premier ordre}

\theorem{Proposition}{Condition Nécessaire}{false}{
    Si $f$ admet un extremum local en un point ouvert $x_0$, alors $x_0$ est un point critique.
}

\remark{La réciproque est fausse. Un point critique peut être un point d'inflexion ($f(x)=x^3$) ou un \textbf{point selle} ($f(x,y)=x^2-y^2$).}

\subsection{Conditions du second ordre}

Soit $x_0$ un point critique de $f$. On étudie la nature de $D^2f(x_0)$ (ou de sa matrice Hessienne $H$).

\theorem{Proposition}{Conditions suffisantes}{false}{
    \begin{itemize}
        \item Si $D^2f(x_0)(h,h) > 0$ pour tout $h \neq 0$ ($H$ définie positive), alors $x_0$ est un \textbf{minimum local strict}.
        \item Si $D^2f(x_0)(h,h) < 0$ pour tout $h \neq 0$ ($H$ définie négative), alors $x_0$ est un \textbf{maximum local strict}.
        \item Si $D^2f(x_0)$ change de signe (valeurs propres de signes opposés), alors $x_0$ est un \textbf{point selle}.
        \item Si $D^2f(x_0)$ est dégénérée (au moins une valeur propre nulle) sans changer de signe, on ne peut pas conclure directement.
    \end{itemize}
}

\vocabulary{Un point selle est un point critique qui n'est ni un maximum, ni un minimum.}
