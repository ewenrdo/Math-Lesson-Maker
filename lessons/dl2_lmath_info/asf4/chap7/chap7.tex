\section{Intégrales définies à un paramètre}

Soit $I$ un intervalle de $\mathbb{R}$ et $J = [a,b]$ un segment de $\mathbb{R}$. On considère une fonction $f : I \times J$.\\
On va s'intéresse aux intégrales de la forme :
\[
F : I \to \mathbb{R}, \; x \mapsto \int_a^b f(x,t) dt
\]

\subsection{Continuité et limites de $F$}

\theorem{Théorème}{Continuité des intégrales définies à un paramètre}{false}{
    Soit $I$ un intervalle de $\mathbb{R}$ et $J = [a,b]$ un segment de $\mathbb{R}$. Soit une fonction $f : I \times J$.\\    
    Si $f$ est continue, alors $F : I \to \mathbb{R}$ telle que $x \mapsto \int_a^b f(x,t) dt$ est bien définie et continue sur $I$.
}

\noindent{\textbf{Démonstration :}\\
    Soit $x\in I$, la fonction $t\mapsto f(x, t)$ est continue sur un segment. Donc, $\int_a^bf(x, t) dt$ est bien définie.\\
    Soit $x_0\in I$, on veut démontrer que $x\mapsto \int_a^bf(x, t) dt$ est continue en $x_0$.\\
    Soit $x_0\in I'\subseteq I$, un segment fermé qui contient $x_0$.  \\
    (Si $I= ]a, b[$ où $[a, b]$, $a<x_0<b$; $I'=[x-\varepsilon, x+\varepsilon]$ assez petit, $I'\subseteq I$. Si $I=[a, b[$, on a $x_0=a$ et $I'=[a, a+\varepsilon]$.)\\
    On a $I'\times J$ compact, donc $f : I'\times J\to \mathbb{R}$ est continue sur un compact. Alors, $f$ est uniformément continue sur $I'\times K$. (Théorème de Heine, cf. Chapitre 11 (11.20)).\\
    Soit $\varepsilon>0$, $\exists\delta>0$ tel que $\forall (x, t), (x', t')\in I'\times J$. Si $|(x, t)-(x', t')|<\delta$, $|f(x, t)-f(x', t')|<\varepsilon$.\\
    Si $|x-x'|<\delta$, alors $|(x, t)-(x', t)|<\delta$, $\forall t\in[a, b]$, et donc, $|f(x, t)-f(x', t)|<\varepsilon$.\\
    Donc, $|F(x)-F(x')|=|\int_a^bf(x, t)-\int_a^bf(x', t)|=|\int_a^bf(x, t)-f(x', t)|$\\
                        $\le \int_a^b|f(x, t)-f(x', t)| dt<\int_a^b \varepsilon dt = \varepsilon (b-a)$.\\
    Donc, on a démontrer que $\forall\varepsilon>0, \exists\delta>0$, tel que si $|x-x'|<\delta$, alors $|F(x)-F(x')|<\varepsilon(b-a)$. Donc $f$ est continue.
}
\reminder{
    $f(x, \cdot)$ converge uniformément vers $h$ sur $J$ signifie qu'il existe une fonction $\alpha$ définie sur $I$ telle que $\alpha(x) \to 0$ lorsque $x \to x_0$ et pour tout $t \in J$, $|f(x,t) - h(t)| \leq \alpha(x)$.
}

\theorem{Théorème}{Limites des intégrales définies à un paramètre}{false}{
    Soit $I$ un intervalle de $\mathbb{R}$ et $J = [a,b]$ un segment de $\mathbb{R}$. On considère une fonction $f : I \times J$.\\    
    Supposons que, étant donné $x_0 \in \overline{I}$, on a :
    \begin{itemize}
        \item $f$ est continue,
        \item $lim_{x \to x_0} f(x,t) = h(t)$ en supposant que la convergence de $f(x, \cdot)$ est \textbf{uniforme} sur $J$.
    \end{itemize}

    Alors $h$ est continue et on a l'intervertibilité de l'intégrale et de la limite : \[lim_{x \to x_0} F(x) = \int_a^b h(t) dt\].
}

\noindent{\textbf{Démonstration :}\\
    Posons $f_n(t) = f(n, t)$ une suite de fonction continue qui converge vers $h(t)$ uniformément. Donc $h$ est continue.\\
    $|\int_a^bf(x, t)dt-\int_a^bh(t) dt| = |\int_a^bf(x, t)-h(t) dt|\le \int_a^b|f(x, t)-h(t)| dt$\\
            $\le \int_a^b \|f(x, t)-h(t)\|_{\infty, J} dt = \|f(x, t)-h(t)\|_{J, \infty}(b-a)$.\\
    Donc, $lim_{x\to+\infty}\int_a^b f(x, t) dt = \int_a^b h(t) dt$.
}

\subsection{Dérivabilité de $F$}

\theorem{Théorème}{Dérivation sous le signe intégral}{false}{
    Soit $I$ un intervalle de $\mathbb{R}$ et $J = [a,b]$ un segment de $\mathbb{R}$. On considère une fonction $f : I \times J$.\\    
    Si $\frac{\partial f}{\partial x}$ existe et est continue sur $I \times J$, alors $F$ est de classe $\mathcal{C}^1$ sur $I$ et pour tout $x \in I$, on a : \[F'(x) = \int_a^b \frac{\partial f}{\partial x}(x,t) dt\]
}

\noindent{\textbf{Démonstration : }\\
    On peut supposer que $I$ est un intervalle fermé et borné.\\
    Soit $h>0$, on veut calculer $lim_{h\to0}\frac{F(x+h)-F(x)}{h}$.\\
    On a $\frac{F(x+h)-F(x)}{h}=\int_a^b \frac{f(x+h)-f(x)}{h}$.
    Posons $G(x, h, t) = \begin{cases}
        \frac{f(x+h)-f(x)}{h},  h\ne0\\ \frac{\partial f}{\partial x}(x, t), h =0
    \end{cases}$. \\
    $G$ est continue car $\partial f/\partial x : I\times J\to \mathbb{R}$ est continue.\\
    Donc, par le théorème précédent, $lim_{h\to0}\int_a^bG(x, h, t) dt = \int_a^b lim_{h\to0} G(x, h, t) dt$ et $lim_{h\to0}\frac{\partial f}{\partial x}(x, t) dt$. \\
    Donc, on a démontré que $F$ est dérivable et $F'(x)=\int_a^b \frac{\partial f}{\partial x} (x, t) dt$.
    Comme $\partial f/\partial x$ est continu par le théorème sur la continuité des intégrales à un paramètre, $x\mapsto \int_a^b \frac{\partial f}{\partial x}(x, t) dt$ est continue. Donc $F$ est $C^1$.
}

\section{Intégrales généralisées à un paramètre}

Soit $I$ un intervalle de $\mathbb{R}$. On considère une fonction $f : I \times [0,+\infty[$ continue.\\
On s'intéresse aux propriétés de la fonction $F : I \to \mathbb{R}$ définie, sous réserve d'existence, par :
\[
    F : I \to \mathbb{R}, \; x \mapsto \int_0^{+\infty} f(x,t) dt
\]

On adapte sans difficulté les énoncés suivants à des intervalles différents de $[0,+\infty[$.
\subsection{Convergence de l'intégrale}
\definition{
    L'intégrale $F(x) = \int_0^{+\infty} f(x,t) dt$ est dite \textbf{simplement convergente} sur $I$ si pour tout $x \in I$, l'intégrale $\int_0^{+\infty} f(x,t) dt$ converge.
}

\definition{
    L'intégrale $F(x) = \int_0^{+\infty} f(x,t) dt$ est dite \textbf{normalement convergente} sur $I$ s'il existe une fonction $g : [0,+\infty[ \to \mathbb{R}_+$ continue telle que pour tout $x \in I$ et tout $t \in [0,+\infty[$, on a $|f(x,t)| \leq g(t)$ et l'intégrale $\int_0^{+\infty} g(t) dt$ converge.
}
\vocabulary{Si une telle fonction existe, on dit que $g$ est une \textbf{fonction de domination} de $f$.}
\remark{
    Si on a la convergence normale sur $I$, alors en particulier l'intégrale converge absolument pour tout $x \in I$, et donc elle converge simplement.
}

\theorem{Lemme}{}{false}{
    Supposons $F(x)$ converge normalement sur $I$. Pour tout $n \in \mathbb{N}$, soit $\varphi_n : I \to \mathbb{R}$ définie par $\varphi_n(x) = \int_0^n f(x,t) dt$. Alors la suite de fonctions $(\varphi_n)_{n \in \mathbb{N}}$ converge uniformément vers $F$ sur $I$.
}
\ndlr{cf. Laurent pour la preuve}

\subsection{Continuité et limites de $F$}
\theorem{Théorème}{Continuité des intégrales généralisées à un paramètre}{false}{
    Supposons que $f : I \times [0,+\infty[ \to \mathbb{R}$ est continue et que $F(x) = \int_0^{+\infty} f(x,t) dt$ converge normalement sur $I$. Alors $F : I \to \mathbb{R}$ est continue sur $I$.
}
\ndlr{cf. Laurent pour la preuve}

\theorem{Limite des intégrales généralisées à un paramètre}{}{false}{
    Soit $I$ un intervalle de $\mathbb{R}$ et $x_0 \in \overline{I}$. Considérons une fonction $f : I \times [0,+\infty[ \to \mathbb{R}$ étant \textbf{continue} sur $I \times [0,+\infty[$.\\
    Supposons que :
    \begin{itemize}
        \item $F(x)$ converge normalement sur $I$,
        \item $\forall t \in [0,+\infty[$, $h(t) := lim_{x \to x_0} f(x,t)$ existe,
        \item cette convergence est uniforme en $t$ sur chaque intervalle de la forme $[0,A]$ avec $A > 0$.
    \end{itemize}
    Alors $h$ est continue d'intégrale $\int_0^{+\infty} h(t) dt$ absolument convergente et on a : \[lim_{x \to x_0} F(x) = \int_0^{+\infty} h(t) dt\].
}
\ndlr{cf. Laurent pour la preuve}

\subsection{Dérivabilité de $F$}

\theorem{Théorème}{Dérivabilité des intégrales généralisées à un paramètre}{false}{
    Soit $I$ un intervalle de $\mathbb{R}$. Considérons une fonction $f : I \times [0,+\infty[ \to \mathbb{R}$.\\
    Supposons que :
    \begin{itemize}
        \item $F(x)$ converge simplement sur $I$,
        \item $f$ a une dérivée partielle par rapport à $x$, qui est continue sur $I \times [0,+\infty[$,
        \item l'intégrale $\int_0^{+\infty} \frac{\partial f}{\partial x}(x,t) dt$ converge normalement sur $I$.
    \end{itemize}
    Alors $F$ est dérivable sur $I$ et pour tout $x \in I$, on a : \[F'(x) = \int_0^{+\infty} \frac{\partial f}{\partial x}(x,t) dt\].
}
\ndlr{cf. Laurent pour la preuve}
